N_rep = 1000

sample_means<-rep(NA,N_rep)
for (N in c(1,2,10,50, 1000))
    {
        for(i in 1:N_rep)
        {
            sample<-runif(N,0,1)
            sample_means[i]<-mean(sample)
        }
    hist(sample_means, freq=F, xlim=c(0,1), main=N)
    abline(v=0.5,col='red')
    }
